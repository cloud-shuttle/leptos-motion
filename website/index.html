Installation

# Add to your Cargo.toml
[dependencies]
leptos = "0.7"
leptos-motion = "0.1.0-alpha"

# Basic usage
use leptos::prelude::*;
use leptos_motion::*;

#[component]
fn MyComponent() -> impl IntoView {
    view! {
         AnimationValue::Number(1.1)
            )
            while_hover=motion_target!(
                "scale" => AnimationValue::Number(1.2)
            )
        >
            "Hover me!"
        
    }
}
